---- ONLY ONCE TO CREATE THE PROJECT -----

# create a venv
python3 -m venv venv

# activate venv
mac/linux: source venv/bin/activate
windows: venv\Scripts\activate


# install django
python3 -m pip install django whitenoise pillow


# create the project
django-admin startproject config .
